from collections import Counter
from nltk import ngrams
from nltk.corpus import stopwords

stopwords = stopwords.words("russian")

bigrams_count = Counter()

for _, row in tqdm(df.iterrows()):
    bigrams = ngrams([w for w in row['lemmas'].split() if w not in stopwords], 2)
    bigrams_count.update(bigrams)

plt.figure(figsize=(15,7))

labels = ['_'.join(b[0]) for b in bigrams_count.most_common(15)]
data = [b[1] for b in bigrams_count.most_common(15)]

plt.title('Bigrams on Meduza', fontsize=20)

plt.xticks(fontsize=16, rotation=50)

plt.bar(np.arange(len(labels)), data, color='green', alpha=0.6, tick_label=labels)
plt.show()

def conv(m):
    converter = {'1': 'Jan',
                 '2': 'Feb',
                 '3': 'Mar',
                 '10': 'Oct',
                 '11': 'Nov',
                 '12': 'Dec'}
    return converter[str(m)]

df['M'] = df['M'].apply(conv)
df = df.groupby(['Y', 'M']) \
       .agg({'T':'mean'}) \
       .unstack()['T'] \
       [['Oct', 'Nov', 'Dec', 'Jan', 'Feb', 'Mar']]
